助教：一键登记提交和成绩

发表于2021-06-16|更新于2022-11-29|杂技python
|字数总计:470|阅读时长:2分钟|阅读量:

做助教工作的时候积累的代码，可以一键登记作业是否提交、作业的得分。学生提交作业时格式化命名文件，根据学号进行匹配，可以大大提高工作效率。

import os
import re
import pandas as pd
import xlwt
folder = '/Users/juncker/Desktop' # 路径
# folder = 'C:/Users/Administrator/Desktop/' # 路径
excel_file = folder+'Result.xls' #统计结果表格
# workbook = xlwt.Workbook()       #新建工作簿
# sheet1 = workbook.add_sheet('是否提交',cell_overwrite_ok=True)          #新建sheet
# workbook.save(excel_file)   #保存


excel_file1 = folder+'考勤表.xlsx' #考勤表表格
print(excel_file1)
df1 = pd.read_excel(excel_file1,header = 0)
#定义一些通用名称,后续任务可以替换
student_id = '学号'
student_name = '姓名'
project = '作业1'
df1 = pd.DataFrame({student_id:df1[student_id],student_name:df1[student_name],project:df1[project]})
l = len(df1[student_id]) #名单长度


#登记是否提交代码
######################################################
#项目概述
#文档中文件命名格式为'学号_姓名',本代码的目的是登记是否提交
######################################################
for doc_name in os.listdir(folder):
    full_path = folder + doc_name  ## doc_name文件名，full_path完整的路径
    if '.xlsx'  in doc_name or 'xls' in doc_name:
        continue  ##判断语句的目的是不统计xlsx文件
    id = doc_name[0:10]
    id = int(id)
    for i in range(l):        
        if id == df1[student_id][i]:
            # print(id)
            df1[project][i] = 1
df1.to_excel(excel_file,sheet_name='是否提交',index=False,header=True)#写入excel文件

import openpyxl
import os
import re
from win32com.client import Dispatch

path = "自定义"
os.chdir(path)  #用于改变当前工作目录到指定的路径

for province in os.scandir(path):  #os.scandir遍历目录
    if province.is_dir():  #path.is_dir() 判断目录是否存在
        prov_name = province.name
        for city in os.scandir(province.path):
            if city.is_dir():
                # print(city.name)
                city_tmp = re.findall(r'([^（）]+)（', city.name) #正则表达式
                if len(city_tmp) == 0:
                    city_name = city.name
                else:
                    city_name = city_tmp[0]
                city_name = re.sub(r'.*?(省)', '', city_name) #re.sub用于替换字符串中匹配项
                city_name = re.sub(r'市$', '', city_name)
                city_name = re.sub(r'自治州$', '', city_name) #正则表达式中，$意为以……结尾
                city_name = re.sub(r'市政府$', '', city_name)
                num = 0
                for root, dirs, files in os.walk(city.path):
                    for file in files:  
                        file_name, file_ext = os.path.splitext(file) #splitext分离文件名与扩展名；默认返回(fname,fextension)元组
                        file_path = os.path.join(root, file)
                        if file_name == ".DS_Store":
                            pass
                        else:
                            if file.endswith(".doc") or file.endswith(".docx") or file.endswith(".txt"): #endswith用于判断字符串是否以指定后缀结尾
                                pdfName = file_name + ".pdf"
                                pdfPath = os.path.join(root, pdfName) 
                                print(file_path)
                                if os.path.exists(pdfPath): #如果新定义的pdf文件名已经在路径存在，就把旧的pdf文件移除
                                    os.remove(pdfPath)
                                else:
                                    pass
                                word = Dispatch("Word.Application")
                                # print(wordPath)
                                word_to_pdf = word.Documents.Open(file_path)
                                word_to_pdf.SaveAs(pdfPath, FileFormat=17)
                                word_to_pdf.Close()
                                # os.remove(file_path)

这段代码做了三件事：

  • 把path目录下所有的文件夹及文件做了一次遍历
  • 把每个文件命名
  • 将word保存为pdf

pywin32

目前网上主流的处理word的方法，是使用一个叫做python-docx的模块，其不足之处在于

  • 功能不完整
  • 功能有缺陷
  • 效果有缺陷

Python模块虽多，也不可能满足所有开发者的需求，而且模块为了便于使用，通常是封装过度的，有些功能无法直接使用，必须直接调用Windows api来实现。完成这一目标有两个办法

  • 用C编写Python扩展模块
  • 编写普通的dll，通过Python的ctypes调用
  • 用pywin32模块：https://github.com/mhammond/pywin32，用python调用VBA的api

pywin32中控制word 的模块，是win32com模块

2.1 启动word、保存

有三种方法：Dispatch、DispatchEx、EnsureDispatch。

  • Dispatch基本命令
  • DispatchEx如果某个word文档进程被占用，Dispatch会报错；DispatchEx多进程
  • EnsureDispatch：内置值要被调用，需要用到一个constants；但是Dispatch和DispatchEx下调用constants会报错
word0 = DispatchEx('Word Application') #启动word程序
word0.Visible = True #显示word程序
#打开已有的word文档：
document0 = word0.Documents.open("Your.docx") #注意路径	
document0.Save() #相当于“文件-保存”
document0.Close(SaveChanges = 0) #关闭文档，参数设置是否保存：0不保存，-1自动保存，-2提示用户

后面绝大多数的操作都以Document对象为起点。

document1 = word0.Documents.Add() #Documents.Add()新建一个Word文档
document1.Range().Text = "您好"
document1.Save() #新建文档，直接按“文件-保存”，将会跳出对话框，输入中文名
document1.SaveAs(FileName = "空白.docx") #直接写个中文名
document1.SaveAs(FileName = "空白.pdf", FileFormat = 17)# 将word另存为pdf

做完所有任务后，建议还是把word程序关掉，不然会吃掉电脑大量资源

word0.Quit() #关闭word程序

下面给一个实例：新建一个文件夹，在文件夹内用代码自动生成50个word文件，并在每个文件内写一定的内容：

from win32com.client import DispatchEx
import os

path = "C:/Users/Administrator/Desktop/"
folder_name = "Py_test"
full_path = path + folder_name + "/"
if not os.path.exists(full_path):
    os.mkdir(full_path) #如果不存在则创建目录

word = DispatchEx('Word.Application') #启动word程序
# word.Visible = True
for i in range(50):
    file = word.Documents.Add() #生成一个word文档
    text = "这是第"+ str(i) +"个word文档"
    file.Range().Text = text
    filename = full_path + "新建文档" + str(i) + ".docx"
    # print(filename)
    file.SaveAs(FileName = filename) #另存为

word.Quit() #关闭word程序

2.3 word文件合并

以2.2中生成的50个word文件为例，下一步是将这些文件内容合并为一个文件。示例代码如下：（存在以下两个待解决问题）

  • 合并的文件顺序如何确定？【用代码中的file变量来控制文件的顺序】
  • 合并后，原有的各种格式是否会出问题？
import win32com.client as win32

word = win32.gencache.EnsureDispatch('Word.Application') #启动word对象应用
word.Visible = False

import os
path = "C:/Users/Administrator/Desktop/Py_test"

#获取目录下所有文件的路径
files = []
for filename in os.listdir(path):
    # print(filename)
    filename = os.path.join(path,filename)
    files.append(filename)
# print(files)

output = word.Documents.Add() #新建空的word文档
for file in files:
    output.Application.Selection.InsertFile(file) #拼接文档

#保存并关闭
output.SaveAs('C:/Users/Administrator/Desktop/合并后的word')
output.Close()

word.Quit() #关闭word程序

from win32com import client
doc_path = "" #必须是绝对路径

word.DisplayAlerts == 0
document = word.Documents.Open(doc_path)

allRange = document.Range() #如果不设置参数，就是选中所有
